44 static int
45 viscii_mbtowc (conv_t conv, ucs4_t *pwc, const unsigned char *s, int n)
46 {
47  unsigned char c = *s;
48  if (c < 0x20)
49  *pwc = (ucs4_t) viscii_2uni_1[c];
50  else if (c < 0x80)
51  *pwc = (ucs4_t) c;
52  else
53  *pwc = (ucs4_t) viscii_2uni_2[c-0x80];
54  return 1;
55 }

106 static int
107 viscii_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
108 {
109  unsigned char c = 0;
110  if (wc < 0x0080 && (wc >= 0x0020 || (0x42100064 & (1 << wc)) == 0)) {
111  *r = wc;
112  return 1;
113  }
114  else if (wc >= 0x00c0 && wc < 0x01b8)
115  c = viscii_page00[wc-0x00c0];
116  else if (wc >= 0x1ea0 && wc < 0x1f00)
117  c = viscii_page1e[wc-0x1ea0];
118  if (c != 0) {
119  *r = c;
120  return 1;
121  }
122  return RET_ILSEQ;
123 }
