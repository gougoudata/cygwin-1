29 static int
30 iso8859_6_mbtowc (conv_t conv, ucs4_t *pwc, const unsigned char *s, int n)
31 {
32  unsigned char c = *s;
33  if (c < 0xa0) {
34  *pwc = (ucs4_t) c;
35  return 1;
36  }
37  else {
38  unsigned short wc = iso8859_6_2uni[c-0xa0];
39  if (wc != 0xfffd) {
40  *pwc = (ucs4_t) wc;
41  return 1;
42  }
43  }
44  return RET_ILSEQ;
45 }

66 static int
67 iso8859_6_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
68 {
69  unsigned char c = 0;
70  if (wc < 0x00a0) {
71  *r = wc;
72  return 1;
73  }
74  else if (wc >= 0x00a0 && wc < 0x00b0)
75  c = iso8859_6_page00[wc-0x00a0];
76  else if (wc >= 0x0608 && wc < 0x0658)
77  c = iso8859_6_page06[wc-0x0608];
78  if (c != 0) {
79  *r = c;
80  return 1;
81  }
82  return RET_ILSEQ;
83 }
