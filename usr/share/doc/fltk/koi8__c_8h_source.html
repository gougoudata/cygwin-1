84 static int
85 koi8_c_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
86 {
87  unsigned char c = 0;
88  if (wc < 0x0080) {
89  *r = wc;
90  return 1;
91  }
92  else if (wc >= 0x00a0 && wc < 0x00a1)
93  c = koi8_c_page00[wc-0x00a0];
94  else if (wc >= 0x0400 && wc < 0x04ef)
95  c = koi8_c_page04[wc-0x0400];
96  else if (wc >= 0x2216 && wc < 0x2217)
97  c = koi8_c_page22[wc-0x2216];
98  if (c != 0) {
99  *r = c;
100  return 1;
101  }
102  return RET_ILSEQ;
103 }
