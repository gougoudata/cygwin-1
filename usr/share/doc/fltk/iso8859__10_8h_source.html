74 static int
75 iso8859_10_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
76 {
77  unsigned char c = 0;
78  if (wc < 0x00a0) {
79  *r = wc;
80  return 1;
81  }
82  else if (wc >= 0x00a0 && wc < 0x0180)
83  c = iso8859_10_page00[wc-0x00a0];
84  else if (wc == 0x2015)
85  c = 0xbd;
86  if (c != 0) {
87  *r = c;
88  return 1;
89  }
90  return RET_ILSEQ;
91 }
