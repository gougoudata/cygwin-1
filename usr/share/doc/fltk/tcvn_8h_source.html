46  if (c < 0x20)

115 static int
116 tcvn_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
117 {
118  unsigned char c = 0;
119  if (wc < 0x0080 && (wc >= 0x0020 || (0x00fe0076 & (1 << wc)) == 0)) {
120  *r = wc;
121  return 1;
122  }
123  else if (wc >= 0x00a0 && wc < 0x01b8)
124  c = tcvn_page00[wc-0x00a0];
125  else if (wc >= 0x0300 && wc < 0x0328)
126  c = tcvn_page03[wc-0x0300];
127  else if (wc >= 0x1ea0 && wc < 0x1f00)
128  c = tcvn_page1e[wc-0x1ea0];
129  if (c != 0) {
130  *r = c;
131  return 1;
132  }
133  return RET_ILSEQ;
134 }
