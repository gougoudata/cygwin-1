77 static int
78 iso8859_13_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
79 {
80  unsigned char c = 0;
81  if (wc < 0x00a0) {
82  *r = wc;
83  return 1;
84  }
85  else if (wc >= 0x00a0 && wc < 0x0180)
86  c = iso8859_13_page00[wc-0x00a0];
87  else if (wc >= 0x2018 && wc < 0x2020)
88  c = iso8859_13_page20[wc-0x2018];
89  if (c != 0) {
90  *r = c;
91  return 1;
92  }
93  return RET_ILSEQ;
94 }
