115 // access to core fonts:
116 // This class provides a "smart pointer" that returns a pointer to an XFontStruct.
117 // The global variable fl_xfont can be called wherever a bitmap "core" font is
118 // needed, e.g. when rendering to a GL context under X11.
119 // With Xlib / X11 fonts, fl_xfont will return the current selected font.
120 // With XFT / X11 fonts, fl_xfont will attempt to return the bitmap "core" font most
121 // similar to (usually the same as) the current XFT font.
122 class Fl_XFont_On_Demand
123 {
124 public:
125  Fl_XFont_On_Demand(XFontStruct* p = NULL) : ptr(p) { }
126  Fl_XFont_On_Demand& operator=(const Fl_XFont_On_Demand& x)
127  { ptr = x.ptr; return *this; }
128  Fl_XFont_On_Demand& operator=(XFontStruct* p)
129  { ptr = p; return *this; }
130  XFontStruct* value();
131  operator XFontStruct*() { return value(); }
132  XFontStruct& operator*() { return *value(); }
133  XFontStruct* operator->() { return value(); }
134  bool operator==(const Fl_XFont_On_Demand& x) { return ptr == x.ptr; }
135  bool operator!=(const Fl_XFont_On_Demand& x) { return ptr != x.ptr; }
136 private:
137  XFontStruct *ptr;
138 };
139 extern FL_EXPORT Fl_XFont_On_Demand fl_xfont;
