75 static int
76 armscii_8_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
77 {
78  unsigned char c = 0;
79  if (wc < 0x0028) {
80  *r = wc;
81  return 1;
82  }
83  else if (wc >= 0x0028 && wc < 0x0030)
84  c = armscii_8_page00[wc-0x0028];
85  else if (wc >= 0x0030 && wc < 0x00a0)
86  c = wc;
87  else if (wc >= 0x00a0 && wc < 0x00c0)
88  c = armscii_8_page00_1[wc-0x00a0];
89  else if (wc >= 0x0530 && wc < 0x0590)
90  c = armscii_8_page05[wc-0x0530];
91  else if (wc >= 0x2010 && wc < 0x2028)
92  c = armscii_8_page20[wc-0x2010];
93  if (c != 0) {
94  *r = c;
95  return 1;
96  }
97  return RET_ILSEQ;
98 }
