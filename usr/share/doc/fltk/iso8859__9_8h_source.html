53 static int
54 iso8859_9_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
55 {
56  unsigned char c = 0;
57  if (wc < 0x00d0) {
58  *r = wc;
59  return 1;
60  }
61  else if (wc >= 0x00d0 && wc < 0x0100)
62  c = iso8859_9_page00[wc-0x00d0];
63  else if (wc >= 0x0118 && wc < 0x0160)
64  c = iso8859_9_page01[wc-0x0118];
65  if (c != 0) {
66  *r = c;
67  return 1;
68  }
69  return RET_ILSEQ;
70 }
